cmake_minimum_required(VERSION 3.10)

function(test)
	list(APPEND TEST_LIST d e f)
	set(TEST_LIST ${TEST_LIST} PARENT_SCOPE)
endfunction()
set(TEST_LIST a b c)
test()
list(LENGTH TEST_LIST TEST_LIST_LENGTH)
message("${TEST_LIST}")
message("TEST_LIST_LENGTH = ${TEST_LIST_LENGTH}")
list(GET TEST_LIST 0 1 2 TEST_LIST_PART)
message("${TEST_LIST_PART}")
list(APPEND TEST_LIST e e f f d d q)
message("${TEST_LIST}")
list(FIND TEST_LIST c TEST_FIND_INDEX)
list(GET TEST_LIST ${TEST_FIND_INDEX} TEST_FIND_VALUE)
message("${TEST_FIND_VALUE}")
list(INSERT TEST_LIST 0 q w e r t y)
message("${TEST_LIST}")
list(REMOVE_ITEM TEST_LIST e)
message("${TEST_LIST}")
list(REMOVE_AT TEST_LIST 0 1 2)
message("${TEST_LIST}")
list(REMOVE_DUPLICATES TEST_LIST)
message("${TEST_LIST}")
list(REVERSE TEST_LIST)
message("${TEST_LIST}")
list(SORT TEST_LIST)
message("${TEST_LIST}")
