project(list)

list(LENGTH mylist LENGTH)
message("list length: ${LENGTH}")
set( mylist ${mylist} d e f)
list(GET mylist 0 1 2 list_values)
message("list values index 0 1 2: ${list_values}")
list(FIND list_value a find_value_index)
message("value a has index: ${find_value_index}")
list(INSERT mylist 0 ${list_values})
message("list after insertion: ${mylist}")
list(REMOVE_ITEM mylist a)
message("list after remove first item: ${mylist}")
list(SORT mylist)
message("list after sitring: ${mylist}")
list(REMOVE_AT mylist 0)
message("list after removing item index 0: ${mylist}")
set(mylist ${mylist} ${mylist})

list(REMOVE_DUPLICATES mylist)
