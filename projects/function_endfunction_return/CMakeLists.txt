cmake_minimum_required(VERSION 3.2.0)
project(func)
function(fff arg)
	message("arg name = ${arg}, arg value = ${${arg}}")
endfunction()
set(VALUE "hello world")
fff(VALUE)
message("variable VALUE hase value=${VALUE}")
function(eee arg)
	set(${arg} "new value")
endfunction()
eee(VALUE)
message("variable VALUE hase value=${VALUE}")
function(hello arg)
	set(${arg} "new value" PARENT_SCOPE)
endfunction()
hello(VALUE)
message("variable VALUE hase value=${VALUE}")
function(ggg arg)
	message("arguments count: ${ARGC}")
	foreach(ARGS ${${ARGV}})
		message("argument value: ${ARGS}")
	endforeach()
endfunction()
list(APPEND ARGLIST "hello" "world" "i am a" "cmake")
ggg(ARGLIST)
